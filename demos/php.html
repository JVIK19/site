<?php
$rota = $_GET['p'] ?? 'home';
switch ($rota) {
  case 'home':
    echo 'Bem-vindo ao site!';
    break;
  case 'sobre':
    echo 'Sobre o projeto.';
    break;
  default:
    http_response_code(404);
    echo 'Página não encontrada';
}
